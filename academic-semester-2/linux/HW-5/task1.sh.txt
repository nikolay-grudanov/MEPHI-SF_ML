
#!/bin/bash

# Запрашиваем имя пользователя и сохраняем его в переменную 'username'
# Ключ -p позволяет вывести строку-приглашение перед вводом 
read -p "Введите имя пользователя: " username

# Выводим информационное сообщение
echo "Ищем информацию для пользователя: $username"

# Используем grep для поиска строки, которая начинается с имени пользователя
grep "^$username:" /etc/passwd
